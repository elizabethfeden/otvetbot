CREATE TABLE facts(
	ID SERIAL PRIMARY KEY NOT NULL,
	GAPLESS_ID INT NOT NULL UNIQUE,
	TITLE TEXT NOT NULL UNIQUE
);
CREATE TABLE chgks(
	ID SERIAL PRIMARY KEY NOT NULL,
	GAPLESS_ID INT NOT NULL UNIQUE,
	AUTHOR TEXT NOT NULL,
	TITLE TEXT NOT NULL UNIQUE,
	TEXT TEXT NOT NULL,
	ANSWER TEXT NOT NULL,
	PUBLISHED BOOLEAN DEFAULT FALSE,
	SEND_STATS BOOLEAN DEFAULT FALSE,
	TRIED INT DEFAULT 0,
	ANSWERED INT DEFAULT 0
);
CREATE TABLE svoyaks(
	ID SERIAL PRIMARY KEY NOT NULL,
	GAPLESS_ID INT NOT NULL UNIQUE,
	AUTHOR TEXT NOT NULL,
	TITLE TEXT NOT NULL UNIQUE,
	TEXT TEXT NOT NULL,
	ANSWER TEXT NOT NULL,
	PUBLISHED BOOLEAN DEFAULT FALSE,
	SEND_STATS BOOLEAN DEFAULT FALSE,
	TRIED INT DEFAULT 0,
	KEYS INT ARRAY[5] DEFAULT ARRAY[]::INT[5]
);
